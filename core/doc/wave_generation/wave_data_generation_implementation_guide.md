# Wave数据生成功能实现指南

## 1. 方案概述

基于调研报告的分析，采用**委托模式**实现独立的 wave 数据生成功能。核心思想是在 `ProjectManager` 中添加 `generate_wave_data()` 方法，通过新建的 `DatasetWaveGenerator` 辅助类来完成具体的波形生成工作。

## 2. 详细实现方案

### 2.1 核心类设计

#### 2.1.1 DatasetWaveGenerator 类

**文件位置：** `core/wave_generator.py`

```python
import os
import logging
from typing import Optional, Dict, List, Any
from .model_engine import ModelEngine
from .data_processing import Dataset_COMP

class DatasetWaveGenerator:
    \"\"\"数据集波形生成器
    
    负责从项目配置中加载数据集并生成波形文件。
    设计原则：最大限度重用现有的数据集加载逻辑。
    \"\"\"
    
    def __init__(self, project_manager):
        \"\"\"
        初始化波形生成器
        
        Args:
            project_manager: ProjectManager 实例
        \"\"\"
        self.project_manager = project_manager
        self.config = project_manager.config
        self.logger = logging.getLogger(__name__)
        
    def generate_wave_data(self, 
                          output_folder: Optional[str] = None,
                          compress: bool = True,
                          force: bool = False) -> Dict[str, Any]:
        \"\"\"
        生成波形数据
        
        Args:
            output_folder: 输出目录，默认为项目目录下的 wave_output
            compress: 是否压缩波形文件
            force: 是否强制覆盖已存在的文件
            
        Returns:
            Dict: 包含生成的文件路径和元数据
            
        Raises:
            ValueError: 配置错误或数据集加载失败
            FileNotFoundError: 数据文件不存在
            PermissionError: 输出目录权限不足
        \"\"\"
        try:
            # 1. 验证和准备输出目录
            output_folder = self._prepare_output_folder(output_folder, force)
            
            # 2. 加载数据集
            dataset = self._load_dataset()
            
            # 3. 生成波形数据
            result = self._generate_wave_files(dataset, output_folder, compress)
            
            # 4. 记录生成信息
            self._log_generation_result(result)
            
            return result
            
        except Exception as e:
            self.logger.error(f\"Wave data generation failed: {e}\")
            raise
    
    def _prepare_output_folder(self, output_folder: Optional[str], force: bool) -> str:
        \"\"\"
        准备输出目录
        
        Args:
            output_folder: 用户指定的输出目录
            force: 是否强制覆盖
            
        Returns:
            str: 验证后的输出目录路径
        \"\"\"
        if output_folder is None:
            output_folder = os.path.join(
                self.project_manager.checkpoint_dir, 
                'wave_output'
            )
        
        # 检查目录是否存在
        if os.path.exists(output_folder):
            if not force:
                # 检查是否有现有文件
                existing_files = self._find_existing_wave_files(output_folder)
                if existing_files:
                    raise FileExistsError(
                        f\"Output directory {output_folder} contains existing wave files: {existing_files}. \"
                        f\"Use force=True to overwrite.\"
                    )
            else:
                self.logger.warning(f\"Force mode: will overwrite files in {output_folder}\")
        
        # 创建目录
        os.makedirs(output_folder, exist_ok=True)
        
        # 验证写入权限
        if not os.access(output_folder, os.W_OK):
            raise PermissionError(f\"No write permission for directory: {output_folder}\")
            
        return output_folder
    
    def _load_dataset(self) -> Dataset_COMP:
        \"\"\"
        加载数据集
        
        复用 ModelEngine 的数据集加载逻辑
        
        Returns:
            Dataset_COMP: 加载的数据集实例
        \"\"\"
        # 创建 ModelEngine 实例来加载数据集
        model_engine = ModelEngine(
            self.project_manager,
            checkpoint_dir=self.project_manager.checkpoint_dir
        )
        
        # 加载数据集
        model_engine.load_dataset(self.config.dataset_type)
        
        # 准备数据（选择特征向量）
        model_engine.prepare_training_data(output_folder=None)  # 不自动生成wave文件
        
        # 返回测试数据集
        return model_engine.dataset_test
    
    def _generate_wave_files(self, 
                           dataset: Dataset_COMP, 
                           output_folder: str, 
                           compress: bool) -> Dict[str, Any]:
        \"\"\"
        生成波形文件
        
        Args:
            dataset: 数据集实例
            output_folder: 输出目录
            compress: 是否压缩
            
        Returns:
            Dict: 生成结果信息
        \"\"\"
        # 调用数据集的 export_to_wave 方法
        file_paths = dataset.export_to_wave(
            output_folder=output_folder,
            description=f\"Wave data for {self.project_manager.project_name} - {self.config.dataset_type}\",
            author=f\"Generated by cli.py\",
            compress=compress
        )
        
        # 构建返回结果
        result = {
            'project_name': self.project_manager.project_name,
            'dataset_type': self.config.dataset_type,
            'output_folder': output_folder,
            'compress': compress,
            'files': file_paths,
            'dataset_info': {
                'magn_list': dataset.magn_list,
                'freq_list': dataset.freq_list,
                'magn_num': dataset.magn_num,
                'freq_num': dataset.freq_num,
                'fs': dataset.fs,
                'time_clipped_s': dataset.time_cliped_s,
                'type': dataset.type
            }
        }
        
        return result
    
    def _find_existing_wave_files(self, output_folder: str) -> List[str]:
        \"\"\"查找已存在的波形文件\"\"\"
        if not os.path.exists(output_folder):
            return []
        
        wave_files = []
        for file in os.listdir(output_folder):
            if file.endswith('.wave'):
                wave_files.append(file)
        
        return wave_files
    
    def _log_generation_result(self, result: Dict[str, Any]):
        \"\"\"记录生成结果\"\"\"
        self.logger.info(f\"Wave data generation completed for project: {result['project_name']}\")
        self.logger.info(f\"Dataset type: {result['dataset_type']}\")
        self.logger.info(f\"Output folder: {result['output_folder']}\")
        self.logger.info(f\"Generated files: {list(result['files'].keys())}\")
        
        # 记录数据集信息
        dataset_info = result['dataset_info']
        self.logger.info(f\"Dataset info: {dataset_info['magn_num']} magnitudes, {dataset_info['freq_num']} frequencies\")
```

# 在 DatasetWaveGenerator 中添加缓存支持
class DatasetWaveGenerator:
    def __init__(self, project_manager):
        self.project_manager = project_manager
        self.config = project_manager.config
        self._dataset_cache = {}
        
    def _load_dataset(self) -> Dataset_COMP:
        \"\"\"加载数据集，支持缓存\"\"\"
        cache_key = f\"{self.config.dataset_type}_{self.config.data_path}\"
        
        if cache_key in self._dataset_cache:
            self.logger.info(f\"Using cached dataset: {cache_key}\")
            return self._dataset_cache[cache_key]
        
        # 加载数据集
        dataset = self._load_dataset_from_config()
        self._dataset_cache[cache_key] = dataset
        
        return dataset
```

### 5.2 批量处理优化

```python
# 在 cli.py 中优化批量处理
def process_wave_generation_batch(project_names, **kwargs):
    \"\"\"批量处理波形生成\"\"\"
    results = []
    failed_projects = []
    
    for project_name in project_names:
        try:
            project_path = f'projects/{project_name}'
            project = ProjectManager(project_path)
            result = project.generate_wave_data(**kwargs)
            results.append(result)
            print(f\"✅ {project_name}: Wave data generated successfully\")
        except Exception as e:
            failed_projects.append((project_name, str(e)))
            print(f\"❌ {project_name}: {e}\")
    
    # 汇总结果
    if results:
        print(f\"\\n📊 Summary: {len(results)} projects succeeded, {len(failed_projects)} failed\")
        
    if failed_projects:
        print(\"\\n❌ Failed projects:\")
        for project_name, error in failed_projects:
            print(f\"   {project_name}: {error}\")
    
    return results, failed_projects
```

### 基本命令
```bash
python cli.py -w PROJECT_NAME
```

### 高级选项
```bash
# 自定义输出目录
python cli.py -w PROJECT_NAME --output /path/to/output

# 禁用压缩
python cli.py -w PROJECT_NAME --no-compress

# 强制覆盖
python cli.py -w PROJECT_NAME -f

# 批量处理
python cli.py -w PROJECT_PATTERN*
```

## 编程接口使用

### 单个项目
```python
from cli import ProjectManager

project = ProjectManager('projects/my_project')
result = project.generate_wave_data()
```

### 批量处理
```python
from cli import ProjectManager

project_names = ['project1', 'project2', 'project3']
for name in project_names:
    project = ProjectManager(f'projects/{name}')
    result = project.generate_wave_data()
```
