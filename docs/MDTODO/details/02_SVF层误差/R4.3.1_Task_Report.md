### 3.2 数据流关键步骤详解

#### 步骤1: 实测数据获取

**输入**: 实测SVF频率响应数据（Excel文件）

**现有基础设施**:
- 文件位置: `exam_data/20251230_SVFNET_SVF_ONLY/20251230_SVF_ONLY.xlsx`
- 读取代码: `src/inference/layers/svf_spice_layer.py` 中的数据读取逻辑

#### 步骤2: 频率响应拟合

**复用代码**: `SVF_ERROR_SIM` 中的拟合逻辑

**输出**: 拟合参数 `{f0, Q, gain}` 或 IIR系数 `{a1, a2, b0, b1, b2}`

#### 步骤3: 参数写入配置

**配置存储方案**:
- 方案A: 直接写入 `config.json`
- 方案B: 单独的参数文件 `measured_svf_params.json`

**推荐方案B**: 分离管理，便于版本控制和参数更新

#### 步骤4: 训练时加载配置

**关键修改点**:
1. `Config` 类解析 `hardware_aware_training` 配置
2. `ModelEngine.build_model()` 传递配置
3. `WaveNet5._init_filters()` 使用实测参数

#### 步骤5: 模型前向传播

**使用实测参数的DIAGIIR层**进行前向传播

#### 步骤6: 反向传播

**仅更新Dense层权重**，SVF层参数固定（trainable=False）

```python
# src/config.py

class Config:
    # ... 现有代码 ...

    def get_hardware_aware_config(self) -> dict:
        """
        获取硬件感知训练配置

        Returns:
            dict: 硬件感知训练配置，包含以下字段:
                - enabled: bool, 是否启用硬件感知训练
                - measured_svf_source: dict, 实测SVF参数来源配置
                - training_mode: dict, 训练模式配置
        """
        hw_config = self.model_subcfg.get('hardware_aware_training', {})

        # 设置默认值
        default_config = {
            'enabled': False,
            'measured_svf_source': {
                'type': 'none',
                'file_path': None
            },
            'training_mode': {
                'trainable_svf': False,
                'trainable_dense': True
            }
        }

        # 合并配置
        return {**default_config, **hw_config}

    def load_measured_svf_params(self) -> dict:
        """
        加载实测SVF参数

        Returns:
            dict: 实测SVF参数，包含各通道的IIR系数
        """
        hw_config = self.get_hardware_aware_config()
        if not hw_config.get('enabled'):
            return None

        source = hw_config.get('measured_svf_source', {})
        source_type = source.get('type', 'none')

        if source_type == 'fitted_params_file':
            file_path = source.get('file_path')
            if file_path and os.path.exists(file_path):
                with open(file_path, 'r') as f:
                    return json.load(f)

        return None
```

```python
class WaveNet5(BaseModel, LayeredModelSupport, SpiceModelSupport):

        """

        return self._init_filters_ideal()

    def _init_filters_ideal(self):
        """
        使用理想参数初始化滤波器（现有逻辑）
        """
        # ... 现有 _init_filters 代码 ...
        pass

    def _init_filters_hardware_aware(self, measured_params: dict):
        """
        使用实测参数初始化滤波器

        Args:
            measured_params: 实测SVF参数字典
        """
        svf_channels = measured_params.get('svf_channels', [])

        # 提取IIR系数
        a1_list = [ch['iir_coefficients']['a1'] for ch in svf_channels]

        self.fast_iir = self.init_iir
```

### 5.1 实测参数加载流程

```python
def load_measured_svf_params_from_file(file_path: str) -> dict:
    """
    从JSON文件加载实测SVF参数

    Args:
        file_path: 参数文件路径

    Returns:
        dict: 解析后的实测参数
    """
    with open(file_path, 'r', encoding='utf-8') as f:
        data = json.load(f)

    # 验证数据格式
    required_keys = ['metadata', 'svf_channels']
    for key in required_keys:
        if key not in data:
            raise ValueError(f"参数文件缺少必要字段: {key}")

    # 验证每个通道的数据
    for i, channel in enumerate(data['svf_channels']):
        if 'iir_coefficients' not in channel:
            raise ValueError(f"通道 {i} 缺少 iir_coefficients")

    return data
```

### 5.2 从拟合结果生成参数文件

```python
def export_fitted_params_to_json(
    fitted_results: list,
    ideal_params: list,
    source_file: str,
    output_path: str,
    fs: int = 2000
) -> None:
    """
    将拟合结果导出为JSON参数文件

    Args:
        fitted_results: 拟合结果列表 [{f0, Q, gain}, ...]
        ideal_params: 理想参数列表 [{f0, Q}, ...]
        source_file: 源数据文件路径
        output_path: 输出JSON文件路径
        fs: 采样率
    """
    from datetime import datetime
    from calibration_analyzer.exam_class import System

    svf_channels = []

    for i, (fitted, ideal) in enumerate(zip(fitted_results, ideal_params)):
        # 计算IIR系数
        iir_coeffs = compute_iir_coefficients(
            f0=fitted['f0'],
            Q=fitted['Q'],
            filter_type=determine_filter_type(i),
            fs=fs
        )

        channel_data = {
            'channel_index': i,
            'filter_type': determine_filter_type(i),
            'ideal_params': {
                'f0': ideal['f0'],
                'Q': ideal['Q']
            },
            'fitted_params': {
                'f0': fitted['f0'],
                'Q': fitted['Q'],
                'gain_db': fitted.get('gain_db', 0.0)
            },
            'iir_coefficients': iir_coeffs
        }
        svf_channels.append(channel_data)

    output_data = {
        'metadata': {
            'source_file': source_file,
            'fitting_date': datetime.now().isoformat(),
            'fitting_method': 'amplitude_only_curve_fit',
            'sample_rate': fs,
            'version': '1.0'
        },
        'svf_channels': svf_channels,
        'summary': compute_summary_statistics(fitted_results, ideal_params)
    }

    with open(output_path, 'w', encoding='utf-8') as f:
        json.dump(output_data, f, indent=2, ensure_ascii=False)
```

### 5.3 IIR系数计算函数

```python
def compute_iir_coefficients(f0: float, Q: float, filter_type: str, fs: int) -> dict:
    """
    根据f0和Q计算IIR滤波器系数

    Args:
        f0: 中心频率 (Hz)
        Q: 品质因数
        filter_type: 滤波器类型 ('highpass', 'bandpass', 'lowpass')
        fs: 采样率

    Returns:
        dict: IIR系数 {a1, a2, b0, b1, b2}
    """
    from calibration_analyzer.exam_class import System
    import numpy as np

    s = System.s
    omega0 = 2 * np.pi * f0

    # 构建传递函数
    if filter_type == 'highpass':
        tf_expr = s**2 / (s**2 + (omega0/Q)*s + omega0**2)
    elif filter_type == 'bandpass':
        tf_expr = (s*(omega0/Q)) / (s**2 + (omega0/Q)*s + omega0**2)
    elif filter_type == 'lowpass':
        tf_expr = omega0**2 / (s**2 + (omega0/Q)*s + omega0**2)
    else:
        raise ValueError(f"未知的滤波器类型: {filter_type}")

    # 创建系统并转换为离散域
    system = System.fromSymbol(tf_expr, f_range=(1, fs/2))

    # 使用 system2params 转换 (复用现有代码)
    from models.wavenet_models import system2params
    params = system2params([system], fs)[0]

    return {
        'a1': params['a1'],
        'a2': params['a2'],
        'b0': params['b0'],
        'b1': params['b1'],
        'b2': params['b2']
    }
```

### 6.1 配置验证函数

```python
def validate_hardware_aware_config(config: dict) -> tuple[bool, list]:
    """
    验证硬件感知训练配置的完整性和正确性

    Args:
        config: 配置字典

    Returns:
        tuple: (是否有效, 错误/警告列表)
    """
    errors = []
    warnings = []

    hw_config = config.get('model_subcfg', {}).get('hardware_aware_training', {})

    if not hw_config.get('enabled', False):
        return True, []

    # 检查必要字段
    if 'measured_svf_source' not in hw_config:
        errors.append("缺少 measured_svf_source 配置")
    else:
        source = hw_config['measured_svf_source']
        if source.get('type') == 'fitted_params_file':
            file_path = source.get('file_path')
            if not file_path:
                errors.append("fitted_params_file 类型需要指定 file_path")
            elif not os.path.exists(file_path):
                errors.append(f"参数文件不存在: {file_path}")

    # 检查训练模式
    training_mode = hw_config.get('training_mode', {})
    if training_mode.get('trainable_svf', False):
        warnings.append("SVF层设置为可训练，这可能导致过拟合")

    is_valid = len(errors) == 0
    return is_valid, errors + warnings
```

### 6.2 参数一致性验证

```python
def validate_measured_params_consistency(
    measured_params: dict,
    model_subcfg: dict
) -> tuple[bool, list]:
    """
    验证实测参数与模型配置的一致性

    Args:
        measured_params: 实测SVF参数
        model_subcfg: 模型子配置

    Returns:
        tuple: (是否一致, 警告列表)
    """
    warnings = []

    # 获取理想配置
    ideal_freqs = model_subcfg.get('init_center_freqs', [])
    ideal_Qs = model_subcfg.get('init_quality_factors', [])

    # 计算预期通道数 (每个频率对应 highpass, bandpass, lowpass)
    expected_channels = len(ideal_freqs) * 3

    svf_channels = measured_params.get('svf_channels', [])
    actual_channels = len(svf_channels)

    if actual_channels != expected_channels:
        warnings.append(
            f"通道数不匹配: 预期 {expected_channels}, 实际 {actual_channels}"
        )

    # 检查频率偏差
    for i, channel in enumerate(svf_channels):
        fitted = channel.get('fitted_params', {})
        ideal = channel.get('ideal_params', {})

        if fitted and ideal:
            f0_deviation = abs(fitted['f0'] - ideal['f0']) / ideal['f0'] * 100
            if f0_deviation > 10:  # 10%阈值
                warnings.append(
                    f"通道 {i} 的f0偏差较大: {f0_deviation:.1f}%"
                )

    is_consistent = len(warnings) == 0
    return is_consistent, warnings
```
