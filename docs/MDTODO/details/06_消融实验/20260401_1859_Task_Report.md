### 2.2 计算流程

**代码位置**: `src/visualization/figure_paper.py:251-262`

```python
def _calculate_parameters(self, paramss):
    """Helper method to calculate wn, zeta, and A from fit parameters."""
    wn = []
    zeta = []
    A = []
    for A_val, B_val, C_val in paramss:
        wn_val = np.sqrt(B_val)              // 角频率 wn = sqrt(B)
        zeta_val = C_val / (2 * wn_val)       // 阻尼比 zeta = C / (2*wn)
        wn.append(wn_val)
        zeta.append(zeta_val)
        A.append(A_val)
    return wn, zeta, A
```

**固有频率计算**:
```python
self.processed_data['fn_origin'] = np.array(wn_origin) / (2 * np.pi)   // fn = wn / (2*pi)
self.processed_data['fn_comped'] = np.array(wn_comped) / (2 * np.pi)
```

### 2.3 漂移计算

**代码位置**: `src/visualization/image_data_process.py:26-42`

```python
def calculate_drift(metric):
    """
    计算参数漂移程度
    
    科学含义: 使用中位数与极值间的最大差值作为漂移指标
    """
    median = metric['median']
    return max(abs(metric['max'] - median), abs(median - metric['min']))
```

**使用方式**: 固有频率漂移需要先从 `processed_data['fn_comped']` 数组计算 min/median/max，再用 `calculate_drift()` 计算漂移。

### 2.4 复用方式

```python
from visualization.figure_paper import ProjectResult

# 加载项目
result = ProjectResult('LSTMu16_base')
result.load_data()
result.process_data()

# 获取固有频率数组 (每个幅度对应一个值)
fn_comped = result.processed_data['fn_comped']

# 计算漂移
fn_drift = max(abs(max(fn_comped) - np.median(fn_comped)), 
               abs(np.median(fn_comped) - min(fn_comped)))
```

## 3. 灵敏度漂移 (Sensitivity Drift)

### 3.1 灵敏度计算

**代码位置**: `src/visualization/figure_paper.py:241-248`

灵敏度 Sn 的计算公式:
```
Sn = A / (4 * π * ζ * fn)
```

其中 A, ζ (zeta), fn 分别来自拟合参数:
- A: 拟合参数中的 A 值
- ζ (zeta): `C / (2 * wn)` 其中 wn = sqrt(B)
- fn: `wn / (2 * π)`

```python
self.processed_data['Sn_origin'] = np.array(
    A_origin) / (4 * np.pi * np.array(zeta_origin) * self.processed_data['fn_origin'])
self.processed_data['Sn_comped'] = np.array(
    A_comped) / (4 * np.pi * np.array(zeta_comped) * self.processed_data['fn_comped'])
```

### 3.2 100Hz 处灵敏度 (插值法)

**代码位置**: `src/visualization/figure_paper.py:205-230`

当 `use_interp=True` 时，通过插值计算 100Hz 处的灵敏度：

```python
def process_data(self, freq_sen=100):
    # ...
    if use_interp:

            sensitive_comped.append(gain_comped_interp)

def _calculate_standard_r2(x, y):
    """计算标准 R²（决定系数）：R² = 1 - SS_res/SS_tot。
    
    使用最小二乘法拟合 y = a*x + b，然后计算拟合优度。
    """
    x = np.asarray(x, dtype=float)
    y = np.asarray(y, dtype=float)
    
    # ... 计算 slope, intercept ...
    
    ss_tot = np.sum((y - y_mean) ** 2)
    ss_res = np.sum((y - y_pred) ** 2)
    
    r_squared = 1.0 - ss_res / ss_tot if ss_tot > 0 else 1.0
    
    return float(r_squared), details
```

### 4.3 按频率点的线性度计算

**代码位置**: `src/visualization/model_analysis.py:14-48`

```python
def _compute_linearity_by_frequency(input_rms_by_freq, output_rms_origin_by_freq, 
                                    output_rms_comped_by_freq, magnitudes, frequencies):
    """按频点计算 R² 线性度"""
    results = []
    
    for i, freq in enumerate(frequencies):
        x = np.array(input_rms_by_freq[i], dtype=float)
        y_origin = np.array(output_rms_origin_by_freq[i], dtype=float)
        y_comped = np.array(output_rms_comped_by_freq[i], dtype=float)
        
        r2_origin, _ = _calculate_standard_r2(x, y_origin)
        r2_comped, _ = _calculate_standard_r2(x, y_comped)
        
        results.append({
            'frequency_hz': float(freq),
            'r_squared_origin': float(r2_origin),
            'r_squared_comped': float(r2_comped),
            'improvement': float(r2_comped - r2_origin),
            # ... 其他数据 ...
        })
    
    return results
```

## 7. 消融实验分析器设计建议

基于现有基础设施，推荐的 `AblationStudyAnalyzer` 设计：

**复用方式**: 直接调用已有函数/类，**不复制代码**

```python
import json
import numpy as np
from visualization.figure_paper import ProjectResult
from visualization.image_data_process import calculate_drift


class AblationStudyAnalyzer:
    """消融实验分析器 - 复用现有基础设施，不复制代码"""
    
    def __init__(self, config: dict):
        self.config = config
        self.projects = self._load_all_projects()
    
    def _load_all_projects(self):
        """使用 ProjectResult 加载所有项目"""
        projects_data = {}
        for proj in self.config['projects']:
            result = ProjectResult(proj['path'])
            result.load_data()
            result.process_data()
            projects_data[proj['name']] = result
        return projects_data
    
    def calculate_natural_frequency_drift(self, project_name: str) -> dict:
        """计算固有频率漂移 - 复用 ProjectResult + calculate_drift()"""
        fn = self.projects[project_name].processed_data['fn_comped']
        
        metric = {
            'min': float(np.min(fn)),
            'max': float(np.max(fn)),
            'median': float(np.median(fn))
        }
        metric['drift'] = calculate_drift(metric)  # 直接调用已有函数
        return metric
    
    def calculate_sensitivity_drift(self, project_name: str, freq_hz: float = 100) -> dict:
        """计算灵敏度漂移 - 复用 ProjectResult.process_data(freq_sen) + calculate_drift()"""
        result = self.projects[project_name]
        result.process_data(freq_sen=freq_hz)  # 100Hz 处灵敏度
        sens = result.processed_data['sensitive_comped']
        
        metric = {
            'min': float(np.min(sens)),
            'max': float(np.max(sens)),
            'median': float(np.median(sens))
        }
        metric['drift'] = calculate_drift(metric)  # 直接调用已有函数
        return metric
    
    def calculate_linearity(self, project_name: str) -> dict:
        """计算平均线性度 (1 - R²) - 直接读取 JSON"""
        path = f'projects/{self.projects[project_name].project_path}/linearity_by_frequency.json'
        with open(path) as f:
            data = json.load(f)
        
        r2_values = [item['r_squared_comped'] for item in data['linearity_by_frequency']]
        nonlinearity = [1 - r2 for r2 in r2_values]
        return {
            'mean': float(np.mean(nonlinearity)) * 100,
            'max': float(np.max(nonlinearity)) * 100,
            'min': float(np.min(nonlinearity)) * 100
        }
    
    def calculate_suppression_rate(self, ref_drift: float, comp_drift: float) -> float:
        """计算抑制率"""
        return (ref_drift - comp_drift) / ref_drift * 100
```
