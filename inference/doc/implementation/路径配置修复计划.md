```python
# 在 _generate_inference_data 方法中添加
def _find_input_file(self, default_path):
    """查找输入文件的多个可能位置"""
    possible_paths = [
        default_path,                              # 用户指定路径
        "temp/dataset_input.wave",                 # 原始期望路径
        "inference/temp/dataset_input.wave",       # 实际文件位置
        os.path.join("inference", "temp", "dataset_input.wave"),
        os.path.join(os.getcwd(), "temp", "dataset_input.wave"),
        os.path.join(os.getcwd(), "inference", "temp", "dataset_input.wave")
    ]
    
    for path in possible_paths:
        if os.path.exists(path):
            return path
    
    # 如果都找不到，返回详细错误信息
    raise FileNotFoundError(
        f"在以下位置都找不到推理输入文件:\n" +
        "\n".join(f"  - {path}" for path in possible_paths) +
        "\n请确保文件存在或通过 config.inference_input_path 指定正确路径。"
    )
```
