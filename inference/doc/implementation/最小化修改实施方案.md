```python
def run_inference_analysis(self):
    """
    运行推理分析，支持数据生成与分析分离
    
    工作流程：
    1. 检查是否已有推理数据
    2. 如果没有，执行推理生成数据
    3. 执行误差分析
    4. 生成可视化报告
    """
    print(f'🔍 推理分析项目: {self.project_name}')
    
    # 初始化推理系统
    from inference import InferenceProcessor
    
    # 检查推理数据
    inference_data_dir = f'{self.checkpoint_dir}/inference'
    os.makedirs(inference_data_dir, exist_ok=True)
    
    # 检查是否已有推理数据
    has_existing_data = self._check_existing_inference_data(inference_data_dir)
    
    if not has_existing_data:
        print("📊 未找到推理数据，开始生成...")
        self._generate_inference_data(inference_data_dir)
        print("✅ 推理数据生成完成")
    else:
        print("📂 发现已有推理数据，跳过生成步骤")
    
    # 执行误差分析
    print("🔬 开始误差分析...")
    analysis_results = self._analyze_inference_errors(inference_data_dir)
    
    # 生成分析报告
    self._generate_analysis_report(analysis_results, inference_data_dir)
    print("✅ 推理分析完成")

def _check_existing_inference_data(self, data_dir):
    """检查是否已有推理数据"""
    required_files = ['nn_layers.wave', 'spice_layers.wave']
    return all(os.path.exists(f'{data_dir}/{file}') for file in required_files)

def _generate_inference_data(self, data_dir):
    """生成推理数据"""
    from inference import InferenceProcessor
    
    # 使用现有的temp/数据或配置中的输入
    input_wave = "temp/dataset_input.wave"
    if not os.path.exists(input_wave):
        input_wave = "temp/dataset_output_original.wave"
    
    # 创建推理处理器
    processor = InferenceProcessor(self.project_path)
    
    # 分层推理（神经网络）
    processor.backend_type = "layer_by_layer"
    processor._initialize_backend("layer_by_layer")
    input_data = processor.load_input_wave(input_wave)
    nn_outputs = processor.backend.infer(input_data, use_scaler=True)
    
    # 保存神经网络分层输出
    combined_nn_data = self._combine_layer_outputs(nn_outputs, "nn")
    processor.wave_processor.save_waveform(combined_nn_data, f'{data_dir}/nn_layers.wave')
    
    # SPICE推理
    processor.backend_type = "spice"
    processor._initialize_backend("spice")
    spice_outputs = processor.backend.infer(input_data, use_scaler=True)
    
    # 保存SPICE分层输出
    combined_spice_data = self._combine_layer_outputs(spice_outputs, "spice")
    processor.wave_processor.save_waveform(combined_spice_data, f'{data_dir}/spice_layers.wave')
    
    # 保存原始输入
    import shutil
    shutil.copy2(input_wave, f'{data_dir}/input.wave')
    
    # 保存推理元数据
    metadata = {
        "project_name": self.project_name,
        "project_path": self.project_path,
        "config": self.config.__dict__,
        "timestamp": self.training_logger._get_timestamp(),
        "input_file": input_wave,
        "num_layers": len(nn_outputs)
    }
    with open(f'{data_dir}/inference_metadata.json', 'w') as f:
        json.dump(metadata, f, indent=2, default=str)

def _analyze_inference_errors(self, data_dir):
    """分析推理误差"""
    from inference import InferenceProcessor
    from calibration_analyzer.waveprocessor import WaveProcessor
    import numpy as np
    
    wave_processor = WaveProcessor()
    
    # 加载数据
    nn_data = wave_processor.load_waveform(f'{data_dir}/nn_layers.wave')
    spice_data = wave_processor.load_waveform(f'{data_dir}/spice_layers.wave')
    
    # 计算逐层误差
    analysis_results = {
        "project_name": self.project_name,
        "timestamp": self.training_logger._get_timestamp(),
        "layer_analysis": []
    }
    
    min_layers = min(len(nn_data.records), len(spice_data.records))
    
    for i in range(min_layers):
        nn_record = nn_data.records[i]
        spice_record = spice_data.records[i]
        
        # 计算误差
        error = nn_record.data - spice_record.data
        
        # 统计分析
        layer_stats = {
            "layer_index": i + 1,
            "mean_error": float(np.mean(error)),
            "std_error": float(np.std(error)),
            "rms_error": float(np.sqrt(np.mean(np.square(error)))),
            "max_error": float(np.max(np.abs(error))),
            "num_samples": error.size,
            "error_shape": error.shape
        }
        
        analysis_results["layer_analysis"].append(layer_stats)
    
    # 保存分析结果
    with open(f'{data_dir}/error_analysis.json', 'w') as f:
        json.dump(analysis_results, f, indent=2)
    
    return analysis_results

def _generate_analysis_report(self, results, data_dir):
    """生成分析报告"""
    # 创建推理处理器进行可视化
    from inference import InferenceProcessor
    
    processor = InferenceProcessor(self.project_path)
    
    # 生成可视化
    try:
        processor.visualizer.compare_layer_with_direct_output(
            f'{data_dir}/input.wave',
            f'{data_dir}/nn_layers.wave',
            f'{data_dir}/spice_layers.wave'
        )
    except Exception as e:
        print(f"可视化生成失败: {e}")
    
    # 打印摘要
    print("\n📈 误差分析摘要:")
    for layer in results["layer_analysis"]:
        print(f"  第{layer['layer_index']}层: RMS={layer['rms_error']:.6f}, "
              f"最大={layer['max_error']:.6f}, 样本数={layer['num_samples']}")

def _combine_layer_outputs(self, layer_outputs, prefix):
    """合并分层输出为单个WaveData"""
    from calibration_analyzer.wavedata import WaveData, WaveRecord
    
    combined_data = WaveData()
    combined_data.description = f"{prefix} layer outputs"
    
    for i, layer_output in enumerate(layer_outputs):
        for j, record in enumerate(layer_output.records):
            # 创建新的记录，添加层信息
            new_record = WaveRecord()
            new_record.data = record.data
            new_record.sample_rate = record.sample_rate
            new_record.channel_names = record.channel_names
            new_record.record_id = f"{prefix}_layer_{i+1}_record_{j+1}"
            new_record.user_metadata = record.user_metadata.copy()
            new_record.user_metadata["layer_index"] = i + 1
            new_record.user_metadata["original_record_id"] = record.record_id
            
            combined_data.records.append(new_record)
    
    return combined_data
```
