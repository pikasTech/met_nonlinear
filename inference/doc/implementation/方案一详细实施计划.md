**详细代码设计**：

```python
"""
运行管理器模块

负责管理仿真运行的生命周期、数据持久化和历史查询
"""

import os
import json
import shutil
from datetime import datetime
import random
import string
from typing import Dict, List, Optional, Any
from pathlib import Path


class RunManager:
    """仿真运行管理器"""
    
    def __init__(self, base_dir: str = "runs"):
        """
        初始化运行管理器
        
        Args:
            base_dir: 运行数据存储的基础目录
        """
        self.base_dir = Path(base_dir)
        self.base_dir.mkdir(exist_ok=True)
        
        # 运行索引文件
        self.index_file = self.base_dir / "runs_index.json"
        self.runs_index = self._load_runs_index()
    
    def _load_runs_index(self) -> Dict[str, Any]:
        """加载运行索引"""
        if self.index_file.exists():
            with open(self.index_file, 'r', encoding='utf-8') as f:
                return json.load(f)
        return {"runs": {}, "last_updated": None}
    
    def _save_runs_index(self):
        """保存运行索引"""
        self.runs_index["last_updated"] = datetime.now().isoformat()
        with open(self.index_file, 'w', encoding='utf-8') as f:
            json.dump(self.runs_index, f, indent=2, ensure_ascii=False)
    
    def create_run(self, project_path: str, config: Dict[str, Any]) -> str:
        """
        创建新的运行记录
        
        Args:
            project_path: 模型项目路径
            config: 运行配置信息
            
        Returns:
            str: 生成的运行ID
        """
        # 生成唯一运行ID
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        random_suffix = ''.join(random.choices(string.ascii_lowercase + string.digits, k=4))
        run_id = f"run_{timestamp}_{random_suffix}"
        
        # 创建运行目录结构
        run_dir = self.base_dir / run_id
        run_dir.mkdir()
        
        # 创建子目录
        subdirs = ["input", "nn_outputs", "spice_outputs", "spice_models", "analysis"]
        for subdir in subdirs:
            (run_dir / subdir).mkdir()
        
        # 创建元数据
        metadata = {
            "run_id": run_id,
            "project_path": project_path,
            "timestamp": datetime.now().isoformat(),
            "config": config,
            "status": "initialized",
            "data_path": str(run_dir),
            "created_by": "InferenceProcessor",
            "files": {
                "input": None,
                "nn_outputs": [],
                "spice_outputs": [],
                "spice_models": []
            }
        }
        
        # 保存元数据
        metadata_file = run_dir / "metadata.json"
        with open(metadata_file, 'w', encoding='utf-8') as f:
            json.dump(metadata, f, indent=2, ensure_ascii=False)
        
        # 更新索引
        self.runs_index["runs"][run_id] = {
            "project_path": project_path,
            "timestamp": metadata["timestamp"],
            "status": "initialized",
            "data_path": str(run_dir)
        }
        self._save_runs_index()
        
        return run_id
    
    def get_run_dir(self, run_id: str) -> Path:
        """获取运行目录路径"""
        if run_id not in self.runs_index["runs"]:
            raise ValueError(f"运行 {run_id} 不存在")
        return Path(self.runs_index["runs"][run_id]["data_path"])
    
    def get_run_metadata(self, run_id: str) -> Dict[str, Any]:
        """获取运行元数据"""
        run_dir = self.get_run_dir(run_id)
        metadata_file = run_dir / "metadata.json"
        
        if not metadata_file.exists():
            raise FileNotFoundError(f"运行 {run_id} 的元数据文件不存在")
        
        with open(metadata_file, 'r', encoding='utf-8') as f:
            return json.load(f)
    
    def update_run_status(self, run_id: str, status: str, additional_info: Dict[str, Any] = None):
        """更新运行状态"""
        # 更新元数据文件
        metadata = self.get_run_metadata(run_id)
        metadata["status"] = status
        metadata["last_updated"] = datetime.now().isoformat()
        
        if additional_info:
            metadata.update(additional_info)
        
        run_dir = self.get_run_dir(run_id)
        metadata_file = run_dir / "metadata.json"
        with open(metadata_file, 'w', encoding='utf-8') as f:
            json.dump(metadata, f, indent=2, ensure_ascii=False)
        
        # 更新索引
        self.runs_index["runs"][run_id]["status"] = status
        self._save_runs_index()
    
    def list_runs(self, project_path: str = None, status: str = None, 
                  limit: int = None) -> List[Dict[str, Any]]:
        """
        列出运行记录
        
        Args:
            project_path: 按项目路径过滤
            status: 按状态过滤
            limit: 限制返回数量
            
        Returns:
            List[Dict]: 运行记录列表
        """
        runs = []
        for run_id, run_info in self.runs_index["runs"].items():
            # 应用过滤条件
            if project_path and run_info["project_path"] != project_path:
                continue
            if status and run_info["status"] != status:
                continue
            
            # 获取完整元数据
            try:
                metadata = self.get_run_metadata(run_id)
                runs.append(metadata)
            except (FileNotFoundError, json.JSONDecodeError):
                continue
        
        # 按时间戳排序（最新的在前）
        runs.sort(key=lambda x: x["timestamp"], reverse=True)
        
        # 应用数量限制
        if limit:
            runs = runs[:limit]
        
        return runs
    
    def register_file(self, run_id: str, file_type: str, file_path: str):
        """
        注册文件到运行记录
        
        Args:
            run_id: 运行ID
            file_type: 文件类型（input/nn_outputs/spice_outputs/spice_models）
            file_path: 文件路径
        """
        metadata = self.get_run_metadata(run_id)
        
        if file_type == "input":
            metadata["files"]["input"] = file_path
        elif file_type in ["nn_outputs", "spice_outputs", "spice_models"]:
            if file_path not in metadata["files"][file_type]:
                metadata["files"][file_type].append(file_path)
        else:
            raise ValueError(f"不支持的文件类型: {file_type}")
        
        # 保存更新的元数据
        run_dir = self.get_run_dir(run_id)
        metadata_file = run_dir / "metadata.json"
        with open(metadata_file, 'w', encoding='utf-8') as f:
            json.dump(metadata, f, indent=2, ensure_ascii=False)
    
    def cleanup_run(self, run_id: str):
        """清理运行数据（删除运行目录和索引记录）"""
        if run_id not in self.runs_index["runs"]:
            return False
        
        # 删除运行目录
        run_dir = self.get_run_dir(run_id)
        if run_dir.exists():
            shutil.rmtree(run_dir)
        
        # 从索引中移除
        del self.runs_index["runs"][run_id]
        self._save_runs_index()
        
        return True
```

**修改点说明**：
- **新增功能**：运行ID生成、目录管理、元数据存储
- **集成点**：与现有 `InferenceProcessor` 集成
- **数据格式**：标准化的JSON元数据格式
- **兼容性**：独立模块，不影响现有功能

```python
"""
离线误差分析模块

提供基于历史运行数据的离线误差分析功能，支持批量分析和对比
"""

import os
import json
import numpy as np
import matplotlib.pyplot as plt
from typing import List, Dict, Any, Optional, Tuple
from pathlib import Path
import scipy.stats

from calibration_analyzer.waveprocessor import WaveProcessor
from calibration_analyzer.wavedata import WaveData
from .run_manager import RunManager


class OfflineErrorAnalyzer:
    """离线误差分析器"""
    
    def __init__(self, run_manager: RunManager):
        """
        初始化离线分析器
        
        Args:
            run_manager: 运行管理器实例
        """
        self.run_manager = run_manager
        self.wave_processor = WaveProcessor()
        
    def analyze_run(self, run_id: str) -> Dict[str, Any]:
        """
        分析指定运行ID的数据
        
        Args:
            run_id: 运行ID
            
        Returns:
            Dict[str, Any]: 分析结果
        """
        # 验证运行数据
        if not self._validate_run_data(run_id):
            raise ValueError(f"运行 {run_id} 的数据不完整或不匹配")
        
        # 加载数据
        nn_outputs = self._load_nn_outputs(run_id)
        spice_outputs = self._load_spice_outputs(run_id)
        
        if not nn_outputs or not spice_outputs:
            raise ValueError(f"运行 {run_id} 缺少神经网络或SPICE输出数据")
        
        # 计算误差
        layer_errors = self._calculate_layer_errors(nn_outputs, spice_outputs)
        
        # 统计分析
        error_statistics = self._calculate_enhanced_statistics(layer_errors)
        
        # 保存分析结果
        analysis_dir = self.run_manager.get_run_dir(run_id) / "analysis"
        analysis_dir.mkdir(exist_ok=True)
        self._save_analysis_results(run_id, layer_errors, error_statistics, analysis_dir)
        
        return {
            "run_id": run_id,
            "num_layers": len(layer_errors),
            "layer_errors": layer_errors,
            "statistics": error_statistics,
            "analysis_dir": str(analysis_dir)
        }
    
    def batch_analyze(self, run_ids: List[str] = None, 
                      project_path: str = None) -> List[Dict[str, Any]]:
        """
        批量分析多个运行
        
        Args:
            run_ids: 指定的运行ID列表，如果为None则分析所有完成的运行
            project_path: 按项目路径过滤
            
        Returns:
            List[Dict[str, Any]]: 分析结果列表
        """
        if run_ids is None:
            # 获取所有已完成的运行
            all_runs = self.run_manager.list_runs(
                project_path=project_path, status="completed"
            )
            run_ids = [run["run_id"] for run in all_runs]
        
        results = []
        for run_id in run_ids:
            try:
                result = self.analyze_run(run_id)
                results.append(result)
                print(f"✅ 分析完成: {run_id}")
            except Exception as e:
                print(f"❌ 分析失败: {run_id}, 错误: {str(e)}")
                
        return results
    
    def compare_runs(self, run_ids: List[str]) -> Dict[str, Any]:
        """
        对比多个运行的误差
        
        Args:
            run_ids: 要对比的运行ID列表
            
        Returns:
            Dict[str, Any]: 对比分析结果
        """
        comparison_data = {
            "run_ids": run_ids,
            "run_metadata": {},
            "layer_error_comparison": {},
            "statistics_comparison": {},
            "summary_comparison": {}
        }
        
        # 分析每个运行
        run_analyses = {}
        for run_id in run_ids:
            try:
                analysis = self.analyze_run(run_id)
                run_analyses[run_id] = analysis
                
                # 收集元数据
                metadata = self.run_manager.get_run_metadata(run_id)
                comparison_data["run_metadata"][run_id] = {
                    "project_path": metadata["project_path"],
                    "timestamp": metadata["timestamp"],
                    "config": metadata["config"]
                }
                
            except Exception as e:
                print(f"❌ 对比分析失败: {run_id}, 错误: {str(e)}")
                continue
        
        # 执行对比分析
        if len(run_analyses) >= 2:
            comparison_data.update(self._perform_cross_run_comparison(run_analyses))
        
        # 保存对比结果
        self._save_comparison_results(comparison_data)
        
        return comparison_data
    
    def _validate_run_data(self, run_id: str) -> bool:
        """验证运行数据的完整性"""
        try:
            metadata = self.run_manager.get_run_metadata(run_id)
            run_dir = self.run_manager.get_run_dir(run_id)
            
            # 检查必要的目录
            required_dirs = ["nn_outputs", "spice_outputs"]
            for dir_name in required_dirs:
                if not (run_dir / dir_name).exists():
                    return False
            
            # 检查是否有输出文件
            nn_outputs_dir = run_dir / "nn_outputs"
            spice_outputs_dir = run_dir / "spice_outputs"
            
            nn_files = list(nn_outputs_dir.glob("*.wave"))
            spice_files = list(spice_outputs_dir.glob("*.wave"))
            
            return len(nn_files) > 0 and len(spice_files) > 0
            
        except Exception:
            return False
    
    def _load_nn_outputs(self, run_id: str) -> List[WaveData]:
        """加载神经网络输出数据"""
        run_dir = self.run_manager.get_run_dir(run_id)
        nn_outputs_dir = run_dir / "nn_outputs"
        
        wave_files = sorted(nn_outputs_dir.glob("layer_*.wave"))
        outputs = []
        
        for wave_file in wave_files:
            wave_data = self.wave_processor.load_waveform(str(wave_file))
            outputs.append(wave_data)
        
        return outputs
    
    def _load_spice_outputs(self, run_id: str) -> List[WaveData]:
        """加载SPICE输出数据"""
        run_dir = self.run_manager.get_run_dir(run_id)
        spice_outputs_dir = run_dir / "spice_outputs"
        
        wave_files = sorted(spice_outputs_dir.glob("spice_layer_*.wave"))
        outputs = []
        
        for wave_file in wave_files:
            wave_data = self.wave_processor.load_waveform(str(wave_file))
            outputs.append(wave_data)
        
        return outputs
    
    def _calculate_layer_errors(self, nn_outputs: List[WaveData], 
                               spice_outputs: List[WaveData]) -> List[np.ndarray]:
        """计算逐层误差"""
        layer_errors = []
        
        min_layers = min(len(nn_outputs), len(spice_outputs))
        
        for i in range(min_layers):
            nn_data = nn_outputs[i]
            spice_data = spice_outputs[i]
            
            # 确保数据长度匹配
            min_records = min(len(nn_data.records), len(spice_data.records))
            
            errors = []
            for j in range(min_records):
                nn_record = nn_data.records[j]
                spice_record = spice_data.records[j]
                
                # 计算误差
                error = nn_record.data - spice_record.data
                errors.append(error)
            
            if errors:
                layer_error = np.concatenate(errors, axis=0)
                layer_errors.append(layer_error)
        
        return layer_errors
    
    def _calculate_enhanced_statistics(self, layer_errors: List[np.ndarray]) -> Dict[str, Any]:
        """计算增强的误差统计指标"""
        statistics = {}
        
        for layer_idx, error_data in enumerate(layer_errors):
            flat_errors = error_data.flatten()
            
            layer_stats = {
                # 基础统计指标
                "mean_error": float(np.mean(flat_errors)),
                "std_error": float(np.std(flat_errors)),
                "max_error": float(np.max(np.abs(flat_errors))),
                "rms_error": float(np.sqrt(np.mean(np.square(flat_errors)))),
                "min_error": float(np.min(flat_errors)),
                "max_positive_error": float(np.max(flat_errors)),
                "max_negative_error": float(np.min(flat_errors)),
                
                # 分位数统计
                "percentiles": {
                    "p25": float(np.percentile(flat_errors, 25)),
                    "p50": float(np.percentile(flat_errors, 50)),
                    "p75": float(np.percentile(flat_errors, 75)),
                    "p90": float(np.percentile(flat_errors, 90)),
                    "p95": float(np.percentile(flat_errors, 95)),
                    "p99": float(np.percentile(flat_errors, 99))
                },
                
                # 分布特征
                "skewness": float(scipy.stats.skew(flat_errors)),
                "kurtosis": float(scipy.stats.kurtosis(flat_errors)),
                
                # 数据规模
                "num_samples": len(flat_errors),
                "num_channels": error_data.shape[1] if len(error_data.shape) > 1 else 1,
                
                # 信噪比（如果可计算）
                "snr_db": self._calculate_snr_db(error_data) if error_data.size > 0 else None
            }
            
            # 频域分析（如果数据足够）
            if len(flat_errors) >= 64:
                layer_stats["frequency_analysis"] = self._analyze_frequency_error(error_data)
            
            statistics[f"layer_{layer_idx + 1}"] = layer_stats
        
        # 全局统计
        if layer_errors:
            all_errors = np.concatenate([e.flatten() for e in layer_errors])
            statistics["global"] = {
                "total_samples": len(all_errors),
                "global_rms": float(np.sqrt(np.mean(np.square(all_errors)))),
                "global_max": float(np.max(np.abs(all_errors))),
                "error_propagation": self._analyze_error_propagation(layer_errors)
            }
        
        return statistics
    
    def _calculate_snr_db(self, error_data: np.ndarray) -> float:
        """计算信噪比（dB）"""
        try:
            signal_power = np.mean(np.square(error_data))
            if signal_power > 0:
                return float(10 * np.log10(1.0 / signal_power))
            return float('inf')
        except:
            return None
    
    def _analyze_frequency_error(self, error_data: np.ndarray) -> Dict[str, Any]:
        """分析频域误差特征"""
        try:
            # 对每个通道进行FFT分析
            freq_analysis = {}
            
            if len(error_data.shape) == 1:
                # 单通道数据
                fft_data = np.abs(np.fft.fft(error_data))
                freq_analysis["spectral_centroid"] = float(np.mean(fft_data))
                freq_analysis["spectral_energy"] = float(np.sum(fft_data**2))
            else:
                # 多通道数据，分析第一个通道
                fft_data = np.abs(np.fft.fft(error_data[:, 0]))
                freq_analysis["spectral_centroid"] = float(np.mean(fft_data))
                freq_analysis["spectral_energy"] = float(np.sum(fft_data**2))
            
            return freq_analysis
        except:
            return {"error": "频域分析失败"}
    
    def _analyze_error_propagation(self, layer_errors: List[np.ndarray]) -> Dict[str, Any]:
        """分析误差传播特征"""
        if len(layer_errors) < 2:
            return {"layers": len(layer_errors), "propagation": "insufficient_data"}
        
        propagation = {
            "num_layers": len(layer_errors),
            "rms_progression": [],
            "amplification_factors": []
        }
        
        for i, error_data in enumerate(layer_errors):
            rms = float(np.sqrt(np.mean(np.square(error_data.flatten()))))
            propagation["rms_progression"].append(rms)
            
            if i > 0:
                prev_rms = propagation["rms_progression"][i-1]
                amplification = rms / prev_rms if prev_rms > 0 else float('inf')
                propagation["amplification_factors"].append(amplification)
        
        return propagation
    
    def _perform_cross_run_comparison(self, run_analyses: Dict[str, Any]) -> Dict[str, Any]:
        """执行跨运行对比分析"""
        comparison = {
            "cross_run_statistics": {},
            "best_performance": {},
            "worst_performance": {}
        }
        
        # 收集所有运行的RMS误差
        run_rms_errors = {}
        for run_id, analysis in run_analyses.items():
            rms_errors = []
            for layer_key, stats in analysis["statistics"].items():
                if layer_key.startswith("layer_"):
                    rms_errors.append(stats["rms_error"])
            run_rms_errors[run_id] = rms_errors
        
        # 找出最佳和最差性能
        if run_rms_errors:
            avg_rms_by_run = {
                run_id: np.mean(rms_list) 
                for run_id, rms_list in run_rms_errors.items()
            }
            
            best_run = min(avg_rms_by_run.items(), key=lambda x: x[1])
            worst_run = max(avg_rms_by_run.items(), key=lambda x: x[1])
            
            comparison["best_performance"] = {
                "run_id": best_run[0],
                "avg_rms_error": best_run[1]
            }
            comparison["worst_performance"] = {
                "run_id": worst_run[0],
                "avg_rms_error": worst_run[1]
            }
        
        return comparison
    
    def _save_analysis_results(self, run_id: str, layer_errors: List[np.ndarray], 
                              statistics: Dict[str, Any], analysis_dir: Path):
        """保存分析结果"""
        # 保存统计数据
        stats_file = analysis_dir / "error_statistics.json"
        with open(stats_file, 'w', encoding='utf-8') as f:
            json.dump(statistics, f, indent=2, ensure_ascii=False)
        
        # 保存误差数据摘要
        error_summary = {
            "run_id": run_id,
            "num_layers": len(layer_errors),
            "layer_shapes": [err.shape for err in layer_errors],
            "analysis_timestamp": datetime.now().isoformat()
        }
        
        summary_file = analysis_dir / "analysis_summary.json"
        with open(summary_file, 'w', encoding='utf-8') as f:
            json.dump(error_summary, f, indent=2, ensure_ascii=False)
    
    def _save_comparison_results(self, comparison_data: Dict[str, Any]):
        """保存对比分析结果"""
        # 在第一个运行的分析目录中保存对比结果
        if comparison_data["run_ids"]:
            first_run_id = comparison_data["run_ids"][0]
            run_dir = self.run_manager.get_run_dir(first_run_id)
            comparison_file = run_dir / "analysis" / "comparison_results.json"
            
            with open(comparison_file, 'w', encoding='utf-8') as f:
                json.dump(comparison_data, f, indent=2, ensure_ascii=False)
```
